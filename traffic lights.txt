#include <wiringPi.h>
#include <stdio.h>
#define SWITCH 1
#define LED0 2
#define LED1 3
#define LED2 21
#define LED3 22
#define LED4 24
#define LED5 5
#define LED6 6

int main (void) {
if (wiringPiSetup() == -1) {
printf("setup wiringPi failed!");
return 1;
}

pinMode(SWITCH, INPUT);
pinMode(SWITCH, PUD_DOWN);
pinMode(LED0, OUTPUT); 
pinMode(LED1, OUTPUT);
pinMode(LED2, OUTPUT);
pinMode(LED3, OUTPUT);
pinMode(LED4, OUTPUT);
pinMode(LED5, OUTPUT);
pinMode(LED6, OUTPUT);

for(;;) {
	digitalWrite(LED0, LOW);
	digitalWrite(LED1, HIGH);
	digitalWrite(LED2, LOW);
	digitalWrite(LED3, LOW);
	digitalWrite(LED4, HIGH);
	digitalWrite(LED5, LOW);
	digitalWrite(LED6, LOW);

	if (digitalRead(SWITCH) ==1) {

		delay(3000);

		digitalWrite(LED4, LOW); 
		digitalWrite(LED3, HIGH);

		delay(2000);

		digitalWrite(LED3, LOW);
		digitalWrite(LED2, HIGH);
		digitalWrite(LED1, LOW);

		delay(500);

		digitalWrite(LED0, HIGH);

		int a;
		for (a=0;a<5;a++) {

			digitalWrite(LED5, HIGH);
			delay(500);

			digitalWrite(LED6, HIGH);
			delay(500);

			digitalWrite(LED5, LOW);
			delay(500);
	
			digitalWrite(LED6, LOW);
			delay(500);
				}
	
		delay(1000);

		}else {
		}
	}
return 0;
}
